let education_count = 0;
let work_experience_count = 0;

function educationAdd() {
    education_count++;
    document.getElementById("education_container").innerHTML += `
    <div class="education_detail">
        <div>
            <label for="course${education_count}">Name of Board/course</label>
            <input type="text" id="course${education_count}" name="course[]" placeholder="Enter your Course Here">
        </div>

        <div>
            <label for="passing_year${education_count}">Passing Year</label>
            <input type="text" id="passing_year${education_count}" name="passing_year[]" placeholder="Enter Your Passing Year" maxlength=4>
        </div>

        <div>
            <label for="percentage${education_count}">Percentage</label>
            <input type="text" id="percentage${education_count}" name="percentage[]" placeholder="Enter Your Percentage Here">
        </div>
    </div>`
}


function educationDelete() {
    const container = document.getElementById('education_container');
    const rows = document.getElementsByClassName('education_detail');

    if(rows.length > 1) {
        container.removeChild(rows[rows.length-1]);
        education_count--;
    }
    else {
        alert("You Cant Delete the Default Fields")
    }
}

    const container = document.getElementById('work_experience_container');
