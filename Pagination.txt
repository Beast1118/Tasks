pagination.js

const totalPages = document.getElementById('demo').innerText;
let currentPage = parseInt(new URLSearchParams(window.location.search).get('page')) || 1;
const maxPageToShow = 10;


function updatePagination() {
    const pageButtonContainer = document.getElementById('pageButtons');
    pageButtonContainer.innerHTML = '';

    let startPage = Math.max(1, currentPage - Math.floor(maxPageToShow/2));
    let endPage = Math.min(totalPages, startPage + maxPageToShow - 1);

    if(endPage - startPage < maxPageToShow - 1) {
        startPage = Math.max(1, endPage - maxPageToShow +1);
    }

    for(let i=startPage; i<=endPage; i++) {
        const button = document.createElement('button');
        button.classList.add('page-link');
        button.dataset.page = i;
        button.textContent = i;

        button.addEventListener('click', ()=>goToPage(i));

        if(i === currentPage) {
            button.classList.add('active');
        }

        const li = document.createElement('li');
        li.classList.add('page-item');
        li.appendChild(button);

        pageButtonContainer.appendChild(li);
    }
    document.getElementById('prevBtn').disabled = currentPage === 1;
    document.getElementById('nextBtn').disabled = currentPage === totalPages;
}

function goToPage(page) {
    if(page < 1 || page > totalPages) {
        return;
    }

    currentPage = page;

    let url = new URL(window.location);
    url.searchParams.set('page', currentPage);
    window.location.href = url.toString();
    
    updatePagination();
    console.log('Current Page: '+page);
}

document.getElementById('prevBtn').addEventListener('click', () => goToPage(currentPage-1));
document.getElementById('nextBtn').addEventListener('click', () => goToPage(currentPage+1));

updatePagination();

function sortColumn(column) {
    const url = new URL(window.location);

    let order = url.searchParams.get('order') || 'asc';

    order = order === 'asc' ? 'desc' : 'asc';

    url.searchParams.set('sort',column);
    url.searchParams.set('order', order);
    url.searchParams.set('page', 1);

    window.location.href = url.toString();
}



function findSimilar() {
    const first_name = document.getElementById('search-text').value;
    console.log(first_name)
    const url = new URL(window.location);

    url.searchParams.set('search', first_name);

    window.location.href = url.toString();
}



pagination_server.js

const path = require('path');
const mysql = require('mysql2');
const express = require('express');
const app = express()

const db = mysql.createConnection({
    host:'localhost',
    user:'root',
    password:'Dev@1234',
    database:'practice'
})

db.connect((err)=>{
    if(err) console.log("Error Occured: "+err.stack());
    else console.log("Database Successfully Connected: "+db.threadId);
});

app.use(express.static(path.join(__dirname, 'public')));
app.set('view engine', 'ejs');
app.set('views', './views');

app.get('/', (req,resp) =>{
    const page = parseInt(req.query.page) || 1;
    const sort = req.query.sort || 'student_id';
    const order = req.query.order === 'desc' ? 'DESC' : 'ASC'
    const query_order = req.query.order === 'desc' ? 'desc' : 'asc';
    const field = req.query.search;
    const limit = 100;
    const offset = (page-1)*limit;


    const allowedColumn = ['student_id', 'student_name', 'student_city', 'student_dateOfBirth'];
    const sortColumn = allowedColumn.includes(sort) ? sort : 'student_id';


    if (!allowedColumn.includes(sort)) {
    return resp.redirect(`/?page=1&sort=student_id&order=${query_order}`); 
    }

    if (page < 1) {
        return resp.redirect('/?page=1');
    }

    let serach_query = '';
    if(field !== '') {
        serach_query = `where student_name like '%${field}%'`;
    }
    console.log(serach_query)

    db.query(`Select * from student_detail ${serach_query} order by ${sortColumn} ${order} limit ${offset}, ${limit}`, (err,students) =>{
        if(err) {
            return resp.send(err);
        }

        db.query(`select count(*) as total from student_detail`, (err, countResult) =>{
            if(err) {
                return resp.send(err);
            }

            const totalResult = countResult[0].total;
            const totalPages = Math.ceil(totalResult/limit);
            
            
            if (page > totalPages) {
                return resp.redirect(`/?page=${totalPages}`);
            }
            
            resp.render('pagination', {
                students,
                currentPage: page,
                totalPages: totalPages
            });
        });
    });
});

app.listen(4000);
